  <script>
    // Data storage and management
    class DashboardData {
      constructor() {
        this.storageKey = 'rt-detect-dashboard';
        this.data = this.loadData();
        this.initializeData();
      }

      loadData() {
        try {
          const stored = localStorage.getItem(this.storageKey);
          return stored ? JSON.parse(stored) : {};
        } catch (error) {
          console.error('Error loading dashboard data:', error);
          return {};
        }
      }

      saveData() {
        try {
          localStorage.setItem(this.storageKey, JSON.stringify(this.data));
        } catch (error) {
          console.error('Error saving dashboard data:', error);
        }
      }

      initializeData() {
        const today = new Date().toDateString();
        
        if (!this.data.stats) {
          this.data.stats = {
            totalDetections: 1247,
            totalPeople: 86,
            activeCameras: 4,
            activeAlerts: 12,
            todayDetections: 142,
            todayPeople: 12,
            todayAlerts: 3
          };
        }

        if (!this.data.dailyStats) {
          this.data.dailyStats = {};
        }

        if (!this.data.dailyStats[today]) {
          this.data.dailyStats[today] = {
            detections: 0,
            people: 0,
            alerts: 0,
            sessions: 0
          };
        }

        if (!this.data.recentActivity) {
          this.data.recentActivity = [];
        }

        if (!this.data.liveSession) {
          this.data.liveSession = {
            isActive: false,
            startTime: null,
            peopleCount: 0,
            confidence: 0,
            fps: 0
          };
        }

        this.saveData();
      }

      updateLiveSession(data) {
        this.data.liveSession = {
          ...this.data.liveSession,
          ...data,
          lastUpdate: new Date().toISOString()
        };
        
        if (data.peopleCount > 0) {
          this.addActivity(`Live detection: ${data.peopleCount} people detected`, 'info');
        }
        
        this.saveData();
      }

      addDetection(count, confidence) {
        const today = new Date().toDateString();
        
        this.data.stats.totalDetections += count;
        this.data.dailyStats[today].detections += count;
        
        if (!this.data.dailyStats[today].maxConfidence || confidence > this.data.dailyStats[today].maxConfidence) {
          this.data.dailyStats[today].maxConfidence = confidence;
        }

        this.saveData();
        this.updateDisplay();
      }

      addPeople(count) {
        const today = new Date().toDateString();
        
        this.data.stats.totalPeople += count;
        this.data.dailyStats[today].people += count;
        
        this.saveData();
        this.updateDisplay();
      }

      addAlert(message) {
        const today = new Date().toDateString();
        
        this.data.stats.activeAlerts++;
        this.data.dailyStats[today].alerts++;
        
        this.addActivity(message, 'alert');
        this.saveData();
        this.updateDisplay();
      }

      addActivity(message, type = 'info') {
        const activity = {
          id: Date.now(),
          message,
          type,
          timestamp: new Date().toISOString()
        };

        this.data.recentActivity.unshift(activity);
        
        // Keep only last 10 activities
        if (this.data.recentActivity.length > 10) {
          this.data.recentActivity = this.data.recentActivity.slice(0, 10);
        }

        this.saveData();
        this.updateActivityDisplay();
      }

      updateDisplay() {
        // Update stats overview
        document.getElementById('activeCameras').textContent = this.data.stats.activeCameras;
        document.getElementById('activeAlerts').textContent = this.data.stats.activeAlerts;
        document.getElementById('totalDetections').textContent = this.formatNumber(this.data.stats.totalDetections);
        document.getElementById('totalPeople').textContent = this.formatNumber(this.data.stats.totalPeople);

        // Update change indicators
        document.getElementById('camerasChange').textContent = `+${this.data.stats.activeCameras - 3} today`;
        document.getElementById('alertsChange').textContent = `+${this.data.stats.todayAlerts} today`;
        document.getElementById('detectionsChange').textContent = `+${this.data.stats.todayDetections} today`;
        document.getElementById('peopleChange').textContent = `+${this.data.stats.todayPeople} today`;

        // Update live data
        this.updateLiveDataDisplay();
      }

      updateLiveDataDisplay() {
        const today = new Date().toDateString();
        const todayStats = this.data.dailyStats[today] || { detections: 0, people: 0, alerts: 0 };

        // Current session data
        document.getElementById('currentPeople').textContent = this.data.liveSession.peopleCount || 0;
        document.getElementById('currentConfidence').textContent = `${Math.round((this.data.liveSession.confidence || 0) * 100)}%`;
        document.getElementById('currentFPS').textContent = this.data.liveSession.fps || 0;
        
        // Session duration
        if (this.data.liveSession.startTime) {
          const duration = Math.floor((Date.now() - new Date(this.data.liveSession.startTime).getTime()) / 1000);
          document.getElementById('sessionDuration').textContent = `${duration}s`;
        }

        // Today's summary
        document.getElementById('todayDetections').textContent = todayStats.detections;
        document.getElementById('todayAlerts').textContent = todayStats.alerts;
        document.getElementById('todayPeople').textContent = todayStats.people;
        document.getElementById('todayAccuracy').textContent = todayStats.maxConfidence ? 
          `${Math.round(todayStats.maxConfidence * 100)}%` : '0%';
      }

      updateActivityDisplay() {
        const container = document.getElementById('recentActivity');
        const activities = this.data.recentActivity.slice(0, 5); // Show last 5 activities

        container.innerHTML = activities.map(activity => `
          <div class="activity-item">
            <div class="activity-icon">${this.getActivityIcon(activity.type)}</div>
            <div class="activity-details">
              <p>${activity.message}</p>
              <span class="activity-time">${this.formatTime(activity.timestamp)}</span>
            </div>
            <span class="activity-status ${activity.type}">${this.capitalize(activity.type)}</span>
          </div>
        `).join('');

        if (activities.length === 0) {
          container.innerHTML = `
            <div class="activity-item">
              <div class="activity-icon">üìä</div>
              <div class="activity-details">
                <p>No recent activity</p>
                <span class="activity-time">System monitoring started</span>
              </div>
              <span class="activity-status info">Info</span>
            </div>
          `;
        }
      }

      getActivityIcon(type) {
        const icons = {
          alert: '‚ö†Ô∏è',
          warning: 'üîî',
          info: 'üìä',
          success: '‚úÖ'
        };
        return icons[type] || 'üìä';
      }

      formatTime(timestamp) {
        const now = new Date();
        const time = new Date(timestamp);
        const diff = now.getTime() - time.getTime();
        
        if (diff < 60000) return 'Just now';
        if (diff < 3600000) return `${Math.floor(diff / 60000)} minutes ago`;
        if (diff < 86400000) return `${Math.floor(diff / 3600000)} hours ago`;
        return time.toLocaleDateString();
      }

      formatNumber(num) {
        if (num >= 1000000) return (num / 1000000).toFixed(1) + 'M';
        if (num >= 1000) return (num / 1000).toFixed(1) + 'K';
        return num.toString();
      }

      capitalize(str) {
        return str.charAt(0).toUpperCase() + str.slice(1);
      }
    }

    // Initialize dashboard
    const dashboard = new DashboardData();

    // Update date and time
    function updateDateTime() {
      const now = new Date();
      const dateOptions = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
      const timeOptions = { hour: '2-digit', minute: '2-digit', second: '2-digit' };
      
      document.getElementById('current-date').textContent = now.toLocaleDateString('en-US', dateOptions);
      document.getElementById('current-time').textContent = now.toLocaleTimeString('en-US', timeOptions);
    }

    // Simulate live data updates
    function simulateLiveData() {
      if (Math.random() < 0.3) { // 30% chance to update live session
        const liveData = {
          isActive: true,
          startTime: dashboard.data.liveSession.startTime || new Date().toISOString(),
          peopleCount: Math.floor(Math.random() * 20),
          confidence: Math.random() * 0.3 + 0.7, // 0.7 to 1.0
          fps: Math.floor(Math.random() * 10) + 20 // 20-30 FPS
        };
        
        dashboard.updateLiveSession(liveData);
        
        // Occasionally add detections and alerts
        if (Math.random() < 0.2) {
          dashboard.addDetection(1, liveData.confidence);
        }
        
        if (Math.random() < 0.1 && liveData.peopleCount > 10) {
          dashboard.addAlert(`High crowd density detected: ${liveData.peopleCount} people`);
        }
      }
      
      dashboard.updateDisplay();
    }

    // Mobile menu toggle
    document.querySelector('.mobile-menu-toggle').addEventListener('click', function() {
      document.querySelector('.menu').classList.toggle('active');
    });

    // Initialize and start updates
    updateDateTime();
    dashboard.updateDisplay();
    dashboard.updateActivityDisplay();

    // Update time every second
    setInterval(updateDateTime, 1000);

    // Update live data every 3 seconds
    setInterval(simulateLiveData, 3000);

    // Listen for storage events from other tabs
    window.addEventListener('storage', function(e) {
      if (e.key === dashboard.storageKey) {
        dashboard.data = JSON.parse(e.newValue);
        dashboard.updateDisplay();
        dashboard.updateActivityDisplay();
      }
    });

    // Add sample initial activities
    setTimeout(() => {
      dashboard.addActivity('System monitoring initialized', 'info');
      dashboard.addActivity('Live detection session started', 'success');
      dashboard.addActivity('Camera 1 connected successfully', 'info');
    }, 1000);
  </script>
